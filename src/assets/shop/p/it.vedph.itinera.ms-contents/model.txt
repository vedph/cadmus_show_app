- contents (MsContent[]):
  - author (string)
  - claimedAuthor (string)
  - work* (string)
  - ranges (MsLocationRange[]):
    - start (MsLocation):
      - n* (int): sheet number.
      - r (boolean): number is Roman.
      - s (string): suffix, containing 0-2 lowercase letters (e.g. r=recto, v=verso, rv=both).
      - l (int): line number.
    - end (MsLocation)
  - state (string, thesaurus ms-content-states)
  - incipit (string)
  - explicit (string)
  - note (string)
  - units (MsContentUnit[]):
    - label* (string)
    - incipit (string)
    - explicit (string)
